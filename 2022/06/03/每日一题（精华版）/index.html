var consecutiveNumbersSum = function(n) {
    let res = 0;
    let i = 1;
    while (i > 0) {
        const sum = (1 + (i - 1)) * (i - 1) / 2;
        if (n - sum <= 0) break;
        if ((n - sum) % i === 0) {
            res++;
        }
        i++;
    }
    return res;
};
